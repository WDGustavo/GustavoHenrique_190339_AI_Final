//cria a raiz da arvore de comportamento
tree("Root")
	parallel
		//vai repetir a condição VigiaPerseguidor
		repeat mute tree("VigiaPerseguidor")
		//vai repetir a condição de morte
		repeat mute tree("Dead")

//cria a arvore de VigiaPerseguidor
tree("VigiaPerseguidor")
	fallback
		tree("Seek")
		while IsHealthLessThan(20.0)
			//faz a descisão para se não ocorrer a ação setada va para a proxima
			fallback
				//chama a arvore de wander
				tree("LookAround")

//cria a arvore de seguir
tree("Seek")
	//caso veja o player
	while SeePlayer
		//faz uma sequencia
		sequence
			//pega a posição do player
			TargetPlayer
			//persegui o player
			SeekPlayer


//cria a arvore de morte
tree("Dead")
	//faz uma sequencia
	sequence
		//se a vida for menor que 1
		IsHealthLessThan(0.1)
		//explode
		Explode

//cria a arvore de olhar aos arredores
tree("LookAround")
	//fala que é uma sequencia
	sequence
		//randomiza
		random(5,1,1,1) 
			Turn(90.0)
			Turn(-90.0)
			Turn(145.0)
			Turn(-27.0)
		//olha para a posição escolhida
		LookAtTarget
		//espera o random
		WaitRandom(0.5,0.1)
